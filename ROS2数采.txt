1. 构建数据采集实验环境
在开始数据采集之前，一个结构清晰、依赖明确的工作空间是必不可少的。虽然你可能已经创建过工作空间，但为了数据采集的完整性和可复现性，我们需要一个包含数据发布节点的定制化环境。

1.1 创建专用的数据采集工作空间
首先，我们创建一个独立的工作空间，避免与现有项目 冲突。我习惯将所有实验性项目放在~/ros2_labs目录
mkdir -p ~/ros2_labs/data_acquisition_ws/src
cd ~/ros2_labs/data_acquisition_ws/src
接下来，我们需要一个能够模拟发布多种传感器数据的Package 。这里，我们不从零开始写C++或Python节点，而是利用ROS2提供的demo_nodes_cpp包，它内置了多种数据发布器，非常适合我们的教学场景。同时，我们也会创建一个简单的自定义包，用于后续扩展。
    # 从ROS2仓库克隆示例包（以Humble版本为例）
    git clone -b humble https://github.com/ros2/demos.git
    # 进入工作空间根目录编译
    cd ~/ros2_labs/data_acquisition_ws
    colcon build --packages-select demo_nodes_cpp
编译成功后，别忘记source环境，这是很多初学者容易遗漏导致命令找不到的关键一步
    source install/setup.bash
1.2 启动模拟数据发布节点
现在，让我们启动几个模拟传感器和状态的发布者。我们将同时运行三个节点：
    模拟激光雷达扫描：发布sensor_msgs/msg/LaserScan类型的话题。
    模拟惯性测量单元：发布sensor_msgs/msg/Imu类型的话题。
    模拟机器人里程计：发布nav_msgs/msg/Odometry类型的话题。
打开第一个终端，运行以下命令：
  # 终端1：发布激光雷达数据
  ros2 run demo_nodes_cpp talker --ros-args -r chatter:=/scan -p message_type:=sensor_msgs/msg/LaserScan
这里使用了--ros-args进行话题重映射（-r），将默认的chatter话题重映射到机器人领域标准的/scan话题。-p参数则指定了发布的消息类型。
打开第二个终端，同样source环境后运行：
  # 终端2：发布IMU数据
  ros2 run demo_nodes_cpp talker --ros-args -r chatter:=/imu/data -p message_type:=sensor_msgs/msg/Imu
打开第三个终端：
  # 终端3：发布里程计数据
  ros2 run demo_nodes_cpp talker --ros-args -r chatter:=/odom -p message_type:=nav_msgs/msg/Odometry
现在，你的系统里应该有三个节点在持续发布数据。我们可以快速验证一下：
  ros2 topic l

void FetchData(char *data, int usLength)
{
    char *pData_head = data;
    while (usLength >= 11)
    {
        if (pData_head[0] != 0x55)
        {
            pData_head++;
            continue;
        }
        switch (pData_head[1])
        {
        case 0x50: //time
            memcpy(&stcTime, &pData_head[2], 8);
            break;
        case 0x51: //acceleration
            memcpy(&stcAcc, &pData_head[2], 8);
            acc.x = stcAcc.a[0] / 32768.00 * 16 * 9.8;
            acc.y = stcAcc.a[1] / 32768.00 * 16 * 9.8;
            acc.z = stcAcc.a[2] / 32768.00 * 16 * 9.8;
            break;
        case 0x52: //gyroscope
            memcpy(&stcGyro, &pData_head[2], 8);
            gyro.x = stcGyro.w[0] / 32768.00 * 2000 / 180 * 3.1415926;
            gyro.y = stcGyro.w[1] / 32768.00 * 2000 / 180 * 3.1415926;
            gyro.z = stcGyro.w[2] / 32768.00 * 2000 / 180 * 3.1415926;
            break;
        case 0x53: //angles
            memcpy(&stcAngle, &pData_head[2], 8);
            angle.r = stcAngle.Angle[0] / 32768.00 * 3.1415926;
            angle.p = stcAngle.Angle[1] / 32768.00 * 3.1415926;
            angle.y = stcAngle.Angle[2] / 32768.00 * 3.1415926;
            break;
        case 0x54: //magnet
            memcpy(&stcMag, &pData_head[2], 8);
            mag.x = stcMag.h[0];
            mag.y = stcMag.h[1];
            mag.z = stcMag.h[2];
            break;
        case 0x59: //quaternion
            memcpy(&stcQuat, &pData_head[2], 8);
            quat.w = stcQuat.q[0] / 32768.00;
            quat.x = stcQuat.q[1] / 32768.00;
            quat.y = stcQuat.q[2] / 32768.00;
            quat.z = stcQuat.q[3] / 32768.00;
            break;
        }
        usLength -= 11;
        pData_head += 11;
    }
}
class WitImuNode : public rclcpp::Node

public:
    WitImuNode()
    : Node("wit_imu_node")
    {
        
        int output_hz = 100;
        timer_ms = output_hz / 1000
        port = "/dev/ttyUSB0"
        baudrate = 115200
        try
        {
            ser.setPort(port);
            ser.setBaudrate(baudrate);
            serial::Timeout to = serial::Timeout::simpleTimeout(500);
            ser.setTimeout(to);
            ser.open();
        }
        catch (serial::IOException &e)
        {
            RCLCPP_INFO(this->get_logger(), "Unable to open port ");
            return;
        }

        if (ser.isOpen())
        {
            RCLCPP_INFO(this->get_logger(), "Serial Port initialized");
        }
        else
        {
            return;
        }

        pub_imu = this->create_publisher<sensor_msgs::msg::Imu>("/imu", 20);
        pub_imu_offline = this->create_publisher<sensor_msgs::msg::Imu>("/imu_offline", 20);
        imu = CJY901();
        
        // ser.flush();
        // int size;
        timer_ = this->create_wall_timer(
            timer_ms, std::bind(&WitImuNode::timer_callback, this));
    }

private:
    void timer_callback()
    {
        int count = ser.available();
        if (count != 0)
        {
            //ROS_INFO_ONCE("Data received from serial port.");
            int num;
            rclcpp::Time now = this->get_clock()->now();
            unsigned char read_buf[count];
            num = ser.read(read_buf, count);
            imu.FetchData((char *)read_buf, num);
            sensor_msgs::msg::Imu imu_data;
            sensor_msgs::msg::Imu imu_offline_data;
            //==============imu data===============
            imu_data.header.stamp = now;
            imu_data.header.frame_id = imu_frame;

            imu_data.linear_acceleration.y = imu.acc.y;

            imu_data.angular_velocity.y = imu.angle.p * 180.0 / M_PI;
            imu_data.angular_velocity.z = imu.angle.y * 180.0 / M_PI;

            tf2::Quaternion curr_quater;
            curr_quater.setRPY(imu.angle.r, imu.angle.p, imu.angle.y); //zyx

            // RCLCPP_INFO(this->get_logger(), "angle: x=%f, y=%f, z=%f",
            //   imu.angle.r, imu.angle.p, imu.angle.y);

            imu_data.orientation.x = curr_quater.x();
            imu_data.orientation.y = curr_quater.y();
            imu_data.orientation.z = curr_quater.z();
            imu_data.orientation.w = curr_quater.w();
            // RCLCPP_INFO(this->get_logger(), "Quaternion: x=%f, y=%f, z=%f, w=%f", 
            //   imu_data.orientation.x, imu_data.orientation.y, imu_data.orientation.z, imu_data.orientation.w);

            //==============imu offline data===============
            imu_offline_data.header.stamp = now;
            imu_offline_data.header.frame_id = imu_frame;

            imu_offline_data.linear_acceleration.x = imu.acc.x;

10.1 点云处理（PCL 集成）
10.1.1 PCL 在 ROS2 中的架构
点云库（Point Cloud Library, PCL）是 3D 感知的核心工具。在 ROS2 中，PCL 通过 pcl_conversions 包与 ROS2 消息系统无缝集成。
ROS2 PointCloud2 消息流：
┌──────────────┐     ┌─────────────┐     ┌──────────────┐
│ 传感器驱动   │────>│ pcl::fromROSMsg │────>│ PCL 处理    │
│(sensor_msgs) │     │              │     │ (滤波/分割)  │
└──────────────┘     └─────────────┘     └──────────────┘
                                               │
                                               ▼
┌──────────────┐     ┌─────────────┐     ┌──────────────┐
│ 下游节点     │<────│ pcl::toROSMsg  │<────│ 处理结果    │
└──────────────┘     └─────────────┘     └──────────────┘
10.1.2 高效点云处理管道设计
典型的点云处理管道包含以下关键步骤：
预处理阶段
    体素格滤波（Voxel Grid Filter）：降采样以减少计算量
    离群点移除（Statistical Outlier Removal）：去除噪声点
    直通滤波（PassThrough Filter）：限定感兴趣区域
特征提取
    法线估计（Normal Estimation）
    关键点检测（Keypoint Detection）
    特征描述子（Feature Descriptors）：FPFH、SHOT等
分割与聚类
    平面分割（RANSAC）
    欧氏聚类（Euclidean Clustering）
    区域生长分割（Region Growing）
10.1.3 点云数据结构优化
点云数据的内存布局对性能影响极大。ROS2 中的 sensor_msgs::msg::PointCloud2 支持两种布局：
    交错布局（Interleaved）：XYZRGBXYZRGB...
    优点：缓存友好，适合逐点处理
    缺点：SIMD 向量化困难
    分离布局（Separated）：XXX...YYY...ZZZ...
    优点：便于 SIMD 优化和 GPU 处理
    缺点：随机访问性能较差
10.1.4 实时点云处理策略
为满足实时性要求，可采用以下优化策略：
    空间索引结构
    KD-Tree：适用于最近邻搜索
    Octree：适用于空间划分和体素化
    R-Tree：适用于范围查询
并行处理
    OpenMP 多线程：CPU 并行
    CUDA 加速：GPU 大规模并行
    TBB（Threading Building Blocks）：任务并行
增量式处理
    滑动窗口：只处理新增数据
    关键帧策略：降低处理频率
    LOD（Level of Detail）：动态调整处理精度

10.2.2 相机标定与畸变校正
相机内参矩阵 K 和畸变系数是图像处理的基础：

\[K = \begin{bmatrix} f_x & 0 & c_x \\ 0 & f_y & c_y \\ 0 & 0 & 1 \end{bmatrix}\]
畸变校正通过以下变换实现：
    径向畸变：$r’ = r(1 + k_1r^2 + k_2r^4 + k_3r^6)$
    切向畸变：考虑镜头与成像平面不平行

核心公式总结
相机投影模型： \(\begin{bmatrix} u \\ v \\ 1 \end{bmatrix} = K \cdot \begin{bmatrix} X/Z \\ Y/Z \\ 1 \end{bmatrix}\)

点云体素化： \(V_{ijk} = \{p \in P : \lfloor p/r \rfloor = (i,j,k)\}\)

传感器融合（卡尔曼滤波）： \(\hat{x}_{k|k} = \hat{x}_{k|k-1} + K_k(z_k - H_k\hat{x}_{k|k-1})\)

ICP 配准目标函数： \(E(R,t) = \sum_{i=1}^{N} ||p_i^{target} - (Rp_i^{source} + t)||^2\)

10.9 常见陷阱与错误（Gotchas）
点云处理陷阱
坐标系混淆
错误：假设所有传感器使用相同坐标系（ROS 坐标系 vs 相机坐标系）
正确：明确每个传感器的坐标定义，使用 tf2 进行转换
点云有序性假设
错误：假设激光雷达点云保持扫描顺序
正确：使用 ring 和 time 字段重建扫描顺序
NaN 和无效点处理
错误：直接处理原始点云，导致算法崩溃
正确：使用 pcl::removeNaNFromPointCloud() 预处理
图像处理陷阱
色彩空间不一致
错误：OpenCV 默认 BGR，ROS 消息是 RGB
正确：使用 cv_bridge 正确转换，注意 encoding 参数
图像步长（stride）忽略
错误：假设图像数据连续存储
正确：使用 step 参数正确索引像素
相机畸变模型误用
错误：对鱼眼相机使用针孔模型
正确：根据相机类型选择合适的畸变模型
融合架构陷阱
时间戳来源不一致
错误：混用系统时间和传感器时间
正确：统一使用同一时钟源，最好是硬件同步
消息队列溢出
错误：队列过小导致消息丢失
正确：根据处理延迟和消息频率合理设置队列大小
外参标定过期
错误：一次标定永久使用
正确：定期验证标定，考虑温度和振动影响
性能优化陷阱
过早优化
错误：一开始就使用复杂的优化技术
正确：先保证正确性，用 profiler 找瓶颈，针对性优化
内存泄漏
错误：GPU 内存只分配不释放
正确：使用 RAII 或智能指针管理 GPU 资源
缓存未命中
错误：随机访问大数组
正确：优化数据布局，提高缓存局部性
